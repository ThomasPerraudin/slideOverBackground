jQuery slideOverBackground Plugin | Demo
This is a demonstration page for the jQuery Plugin slideOverBackground
In your HTML : 
<html>
  <head>
    <link rel="stylesheet" href="slideOverBackground/jquery.slideOverBackground.css" type="text/css" media="all" />
    <script src="js/jquery-1.9.0.min.js"></script>
    <script src="slideOverBackground/jquery.slideOverBackground.js"></script>
  </head>
  <body>
    <div id="mainElement">
      Your contents here.
    <div>
  </body>
</html>

        

          In your JavaScript :
        

        


//slideOverBackground initialization
$("#mainElement").slideOverBackground({
  pictures: ["img/pic1.jpg", "img/pic2.jpg", "img/pic3.jpg"],  //list of background images
  delay: 5000, //delay between transitions (optional)
  transitionTime: 1000 //time of transition (optional)
});
